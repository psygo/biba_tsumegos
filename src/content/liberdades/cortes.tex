\problemAnswerDiagram[%
  sgf folder      = liberdades/cortes,
  sgf filename    = cortes.6,
  problem text    = {Uma técnica excelente para desconexões.},
  answer text one = {\emph{Correto.} Em japonês, este movimento em que se joga entre as pedras adversárias se chama \emph{warikomi}, ou, em inglês, \emph{wedge}. O objetivo é criar dois pontos de corte, A e B, de uma vez só. Note, porém, que, em geral, é preciso muita força externa para que isso funcione.},
  answer text two = {\emph{Incorreto.} Às vezes, isto funciona de maneira parecida com o warikomi, mas, como Branco já possui a boca-de-tigre preparada à esquerda, não conseguiremos cortar depois da conexão em Branco 2 (A ou B também funcionariam).},
]
