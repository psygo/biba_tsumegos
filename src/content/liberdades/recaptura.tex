\chapter{Recaptura}

\emptypage

\problemAnswerDiagram[%
  sgf folder      = liberdades/recaptura,
  sgf filename    = recaptura.1,
  problem text    = {É possível capturar algo branco? Ou é suicídio?},
  answer text one = {\emph{Correto.} Sim, é possível capturar duas pedras brancas. Em seguida, a pedra preta que efetua a captura estará imediatamente em \emph{atari}, isto é, ela possuirá somente 1 liberdade, então Branco terá a opção de uma contra- ou recaptura.},
  answer text two = {\emph{Variação.} Caso Preto opte por não capturar, Branco pode salvar as pedras e deixar Preto com fraquezas no exterior. Isso não quer dizer que não capturar é um erro, pois pode haver outros movimentos mais importantes no tabuleiro.},
]

\problemAnswerDiagram[%
  sgf folder      = liberdades/recaptura,
  sgf filename    = recaptura.4,
  problem text    = {Se forem 3 ou mais pedras, muda alguma coisa?},
  answer text one = {\emph{Correto.} O princípio é o mesmo. Branco pode recapturar, mas Preto acaba se conectando e preserva a iniciativa.},
  answer text two = {\emph{Incorreto.} Se Preto jogar outra coisa, Branco pode jogar em 2 para capturar o grupo preto à direita.},
]

\problemAnswerDiagram[%
  sgf folder      = liberdades/recaptura,
  sgf filename    = recaptura.3,
  problem text    = {Tanto Preto quanto Branco estão em estados terminais.},
  answer text one = {\emph{Correto.} Preto captura, e, depois, Branco pode recapturar, mas isso não muda o status do grupo branco, que é capturado. Mais à frente isto ficará mais claro, isto é basicamente uma corrida de captura do tipo ``olho vs sem olho''.},
  answer text two = {\emph{Incorreto.} Se Branco puder, pode capturar tudo com 2.},
]

\problemAnswerDiagram[%
  sgf folder      = liberdades/recaptura,
  sgf filename    = recaptura.5,
  problem text    = {O extremo canto, com as coordenadas 1-1 e 2-1, é sempre muito peculiar.},
  answer text one = {\emph{Correto.} Branco recaptura, mas este não é o fim da sequência.},
  answer text two = {\emph{Continuação.} Como a recaptura termina no ponto 1-1, Preto pode recapturar a recaptura. Branco acaba com somente 1 olho, o que é insuficiente para viver incondicionalmente no tabuleiro.},
]

\problemAnswerDiagram[%
  sgf folder      = liberdades/recaptura,
  sgf filename    = recaptura.2,
  problem text    = {Branco acaba de jogar a pedra marcada com o objetivo de separar os grupos pretos. Mas funciona?},
  answer text one = {\emph{Correto.} Preto pode utilizar de uma recaptura para impossibilitar o corte.},
  answer text two = {\emph{Incorreto.} Desta maneira, Branco conecta as pedras de corte ao exterior.},
]
